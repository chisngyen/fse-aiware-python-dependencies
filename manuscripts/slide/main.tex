\begin{frame}{Related Work --- 3 hướng tiếp cận}
\scriptsize
\begin{columns}[T]
\column{0.33\textwidth}
\begin{block}{\color{navydeep}1. Knowledge Graph}
\textbf{Ý tưởng:} Encode quan hệ module--version thành đồ thị tri thức từ registry (Libraries.io, PyPI). Truy vấn graph để suy ra dependency tương thích.

\vspace{0.3em}
\textbf{Ưu:} cấu trúc rõ, deterministic, dễ giải thích.

\vspace{0.3em}
\textcolor{alert}{\textbf{Hạn chế:}} DB lỗi thời nhanh, cần cập nhật liên tục, gãy khi registry đổi schema.
\end{block}

\column{0.33\textwidth}
\begin{block}{\color{navydeep}2. Regex / Log-parsing}
\textbf{Ý tưởng:} Parse error log bằng regex patterns để suy ra missing dependency, sau đó patch môi trường rồi build lại.

\vspace{0.3em}
\textbf{Ưu:} lightweight, runtime-driven, không cần DB.

\vspace{0.3em}
\textcolor{alert}{\textbf{Hạn chế:}} \emph{brittle} --- chỉ cần log đổi định dạng (Python/pip update) là parser vỡ.
\end{block}

\column{0.33\textwidth}
\begin{block}{\color{navydeep}3. LLM + RAG}
\textbf{Ý tưởng:} LLM sinh giả thuyết, RAG cung cấp PyPI metadata, Docker build cho feedback, lặp lại.

\vspace{0.3em}
\textbf{Ưu:} adaptive, không cần hạ tầng tĩnh, tận dụng pretrain knowledge.

\vspace{0.3em}
\textcolor{alert}{\textbf{Hạn chế:}} thử-sai mù quáng --- thất bại \emph{không được chuyển thành tri thức} để tỉa không gian.
\end{block}
\end{columns}

\vspace{0.4em}
\centering\footnotesize
\textcolor{mDarkTeal}{\textbf{$\Rightarrow$ Khoảng trống:} chưa có phương pháp nào \emph{học từ thất bại} để thu hẹp không gian tìm kiếm một cách hệ thống.}
\end{frame}

\begin{frame}{CGAR --- Formulation as CSP}
\begin{columns}[T]
\column{0.50\textwidth}
\scriptsize
\textbf{Formulation:} Tuple $\mathcal{P} = \langle X, D, C \rangle$.

\vspace{0.4em}
\textbf{Variables $X$:} các thư viện $\{P_1, \ldots, P_n\}$ + phiên bản Python $\pi$.

\vspace{0.4em}
\textbf{Domains $D$:}
\[
D(P_i) = \{ v \mid \text{req\_py}(P_i, v) \models \pi \land \text{has\_wheel}(P_i, v) \}
\]
Sắp xếp \textit{wheel-first}, semver giảm dần.

\vspace{0.4em}
\textbf{Constraints $C$:}
\begin{itemize}
  \item \textbf{Hard}: cấm vĩnh viễn version lỗi rõ ràng
  \item \textbf{Soft}: cấm tạm nếu lỗi $\ge 2$ lần
  \item \textbf{Upper Bound}: $v_i < \text{ub}(P_i)$ -- cấm cả khoảng
\end{itemize}

\vspace{0.4em}
\textbf{Goal:} tìm $\mathcal{A} = \{\pi^*, v_1^*, \ldots, v_n^*\}$ thỏa $C$; giới hạn $k=50$ attempts.

\column{0.48\textwidth}
\begin{figure}
    \centering
    \includegraphics[width=\linewidth]{cgar_pruning.png}
    \caption{HARD / SOFT / UPPER tỉa cây tìm kiếm $\to$ 1 path khả thi.}
\end{figure}
\end{columns}
\end{frame}
